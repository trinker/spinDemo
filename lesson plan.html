words <- c()
freqs <- c()
position <- 1
x <- cat

for (i in unique(x)) {
  current.word <- which(x==i)
  frequencies <- length(current.word)/length(x)
  freqs[position]<-frequencies
  words[position]<-i
  position<-position + 1
}

# Data Frames
